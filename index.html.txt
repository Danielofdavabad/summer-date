        let speedX = 0, speedY = 0;
        let btnX = 0, btnY = 0;
        let mouseX = window.innerWidth/2, mouseY = window.innerHeight/2;
        
        function getBtnBounds() {
            const rect = noBtn.getBoundingClientRect();
            return { width: rect.width, height: rect.height };
        }
        
        function setBtnPosition(x, y) {
            const bounds = getBtnBounds();
            const maxX = window.innerWidth - bounds.width - 10;
            const maxY = window.innerHeight - bounds.height - 10;
            btnX = Math.min(Math.max(x, 10), maxX);
            btnY = Math.min(Math.max(y, 10), maxY);
            noBtn.style.left = btnX + 'px';
            noBtn.style.top = btnY + 'px';
        }
        
        document.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
            
            const btnRect = noBtn.getBoundingClientRect();
            const centerX = btnRect.left + btnRect.width/2;
            const centerY = btnRect.top + btnRect.height/2;
            const dist = Math.hypot(mouseX - centerX, mouseY - centerY);
            
            if (dist < 150) {
                const angle = Math.atan2(centerY - mouseY, centerX - mouseX);
                const force = Math.min(30, 200 / (dist + 5));
                speedX += Math.cos(angle) * force;
                speedY += Math.sin(angle) * force;
                const maxSpeed = 25;
                speedX = Math.min(maxSpeed, Math.max(-maxSpeed, speedX));
                speedY = Math.min(maxSpeed, Math.max(-maxSpeed, speedY));
            } else {
                speedX *= 0.98;
                speedY *= 0.98;
                if (Math.abs(speedX) < 0.1) speedX = 0;
                if (Math.abs(speedY) < 0.1) speedY = 0;
            }
        });
        
        function animateNoBtn() {
            if (!noBtn || noBtn.style.display === 'none') {
                requestAnimationFrame(animateNoBtn);
                return;
            }
            
            if (speedX !== 0 || speedY !== 0) {
                let newX = btnX + speedX;
                let newY = btnY + speedY;
                const bounds = getBtnBounds();
                const maxX = window.innerWidth - bounds.width - 10;
                const maxY = window.innerHeight - bounds.height - 10;
                
                if (newX <= 10) { newX = 10; speedX = -speedX * 0.6; }
                if (newX >= maxX) { newX = maxX; speedX = -speedX * 0.6; }
                if (newY <= 10) { newY = 10; speedY = -speedY * 0.6; }
                if (newY >= maxY) { newY = maxY; speedY = -speedY * 0.6; }
                
                setBtnPosition(newX, newY);
            }
            requestAnimationFrame(animateNoBtn);
        }
        
        function initNoBtnPosition() {
            const containerRect = initialButtons.getBoundingClientRect();
            const btnRect = noBtn.getBoundingClientRect();
            let leftPos = containerRect.left + containerRect.width - btnRect.width - 20;
            let topPos = containerRect.top + 10;
            if (leftPos + btnRect.width > window.innerWidth - 10) leftPos = window.innerWidth - btnRect.width - 10;
            if (topPos < 10) topPos = 10;
            setBtnPosition(leftPos, topPos);
        }

            }
        });

            }
        });
        
        // ========== منطق تاریخ ==========
        const SUMMER_MONTHS = [4, 5, 6];
        const TARGET_YEAR = 1405;
        
        function populateShamsiDates() {
            const yearSelect = document.getElementById('yearSelect');
            const monthSelect = document.getElementById('monthSelect');
            const daySelect = document.getElementById('daySelect');
            
            yearSelect.innerHTML = '';
            let option = document.createElement('option');
            option.value = TARGET_YEAR;
            option.textContent = TARGET_YEAR;
            option.selected = true;
            yearSelect.appendChild(option);
            
            monthSelect.innerHTML = '';
            for (let i = 0; i < SUMMER_MONTHS.length; i++) {
                let m = SUMMER_MONTHS[i];
                let opt = document.createElement('option');
                opt.value = m;
                opt.textContent = m + ' - ' + getMonthName(m);
                if (m === 4) opt.selected = true;
                monthSelect.appendChild(opt);
            }
            
            function getMonthName(m) {
                const names = ['فروردین','اردیبهشت','خرداد','تیر','مرداد','شهریور','مهر','آبان','آذر','دی','بهمن','اسفند'];
                return names[m-1];
            }
            
            function getDaysInMonth(year, month) {
                if (month <= 6) return 31;
                if (month <= 11) return 30;
                return (year % 33 === 1 || year % 33 === 2 || year % 33 === 3 || year % 33 === 29) ? 30 : 29;
            }
            
            function updateDays() {
                let year = parseInt(yearSelect.value);
                let month = parseInt(monthSelect.value);
                let days = getDaysInMonth(year, month);
                daySelect.innerHTML = '';
                for (let i = 1; i <= days; i++) {
                    let opt = document.createElement('option');
                    opt.value = i;
                    opt.textContent = i;
                    if (i === 15) opt.selected = true;
                    daySelect.appendChild(opt);
                }
            }
            
            monthSelect.addEventListener('change', updateDays);
            updateDays();
        }

        async function sendToFormspree(data) {
            try {
                const formData = new FormData();
                for (const key in data) {
                    formData.append(key, data[key]);
                }
                formData.append('_gotcha', '');
                
                const response = await fetch(FORMSPREE_ENDPOINT, {
                    method: 'POST',
                    headers: { 'Accept': 'application/json' },
                    body: formData
                });
                
                if (response.ok) return true;
                else return fallbackSend(data);
            } catch (error) {
                return fallbackSend(data);
            }
        }
        
        function fallbackSend(data) {
            const form = document.createElement('form');
            form.method = 'POST';
            form.action = FORMSPREE_ENDPOINT;
            form.target = '_blank';
            form.style.display = 'none';
            for (const key in data) {
                const input = document.createElement('input');
                input.type = 'hidden';
                input.name = key;
                input.value = data[key];
                form.appendChild(input);
            }
            document.body.appendChild(form);
            form.submit();
            document.body.removeChild(form);
            return true;
        }

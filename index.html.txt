<script>
    (function() {
        // ------------------- مرحله اول: فرار دکمه نه -------------------
        const noBtn = document.getElementById('noBtn');
        const yesBtn = document.getElementById('yesBtn');
        const initialButtons = document.getElementById('initialButtons');
        const questionDiv = document.getElementById('questionDiv');
        const planPanel = document.getElementById('planPanel');
        
        // توابع فرار
        function moveNoButton() {
            if (!noBtn) return;
            const containerRect = initialButtons.getBoundingClientRect();
            const btnRect = noBtn.getBoundingClientRect();
            const maxLeft = containerRect.width - btnRect.width - 10;
            const maxTop = containerRect.height - btnRect.height - 10;
            let newLeft = Math.random() * maxLeft;
            let newTop = Math.random() * maxTop;
            const yesBtnRect = yesBtn.getBoundingClientRect();
            const relativeYesLeft = yesBtnRect.left - containerRect.left;
            const relativeYesTop = yesBtnRect.top - containerRect.top;
            const distance = Math.hypot(newLeft - relativeYesLeft, newTop - relativeYesTop);
            if (distance < 80) {
                newLeft = (newLeft + 70) % maxLeft;
                newTop = (newTop + 50) % maxTop;
            }
            noBtn.style.position = 'absolute';
            noBtn.style.left = newLeft + 'px';
            noBtn.style.top = newTop + 'px';
            noBtn.style.margin = '0';
            noBtn.classList.add('move-away');
        }
        
        noBtn.addEventListener('mouseenter', moveNoButton);
        noBtn.addEventListener('touchstart', function(e) {
            e.preventDefault();
            moveNoButton();
        });
        noBtn.addEventListener('click', function(e) {
            e.preventDefault();
            moveNoButton();
            noBtn.style.transform = 'rotate(2deg) scale(0.98)';
            setTimeout(() => { noBtn.style.transform = ''; }, 150);
        });
        noBtn.addEventListener('focus', function() { noBtn.blur(); moveNoButton(); });
        
        // وقتی بله زده شد: مرحله دوم رو نشون بده
        yesBtn.addEventListener('click', function() {
            // مخفی کردن دکمه‌ها و سوال اصلی
            initialButtons.style.display = 'none';
            questionDiv.style.display = 'none';
            // نمایش پنل برنامه‌ریزی
            planPanel.style.display = 'block';
            // ریختن مقادیر پیش‌فرض تاریخ شمسی
            populateShamsiDates();
            // یه افکت شاد کوچک
            tinyConfetti();
        });
        
        // ------------------- توابع تاریخ شمسی -------------------
        function populateShamsiDates() {
            const yearSelect = document.getElementById('yearSelect');
            const monthSelect = document.getElementById('monthSelect');
            const daySelect = document.getElementById('daySelect');
            
            const currentYear = 1404; // سال جاری شمسی تقریبی
            for (let i = 1400; i <= 1406; i++) {
                let option = document.createElement('option');
                option.value = i;
                option.textContent = i;
                if (i === currentYear) option.selected = true;
                yearSelect.appendChild(option);
            }
            for (let i = 1; i <= 12; i++) {
                let option = document.createElement('option');
                option.value = i;
                option.textContent = i + ' - ' + getMonthName(i);
                monthSelect.appendChild(option);
            }
            function updateDays() {
                let year = parseInt(yearSelect.value);
                let month = parseInt(monthSelect.value);
                let daysInMonth = getShamsiDaysInMonth(year, month);
                daySelect.innerHTML = '';
                for (let i = 1; i <= daysInMonth; i++) {
                    let opt = document.createElement('option');
                    opt.value = i;
                    opt.textContent = i;
                    if (i === 10) opt.selected = true;
                    daySelect.appendChild(opt);
                }
            }
            function getMonthName(m) {
                const names = ['فروردین','اردیبهشت','خرداد','تیر','مرداد','شهریور','مهر','آبان','آذر','دی','بهمن','اسفند'];
                return names[m-1];
            }
            function getShamsiDaysInMonth(year, month) {
                if (month <= 6) return 31;
                if (month <= 11) return 30;
                // اسفند: معمولاً 29 (کبیسه‌گیری ساده)
                if (isLeapYearShamsi(year)) return 30;
                else return 29;
            }
            function isLeapYearShamsi(year) {
                return (year % 33 === 1 || year % 33 === 2 || year % 33 === 3 || year % 33 === 29);
            }
            yearSelect.addEventListener('change', updateDays);
            monthSelect.addEventListener('change', updateDays);
            updateDays();
        }
        
        // ------------------- ثبت نهایی قرار -------------------
        const submitBtn = document.getElementById('submitPlanBtn');
        const finalSuccessDiv = document.getElementById('finalSuccess');
        
        submitBtn.addEventListener('click', function() {
            const year = document.getElementById('yearSelect').value;
            const month = document.getElementById('monthSelect').value;
            const day = document.getElementById('daySelect').value;
            const time = document.getElementById('timeInput').value;
            const meal = document.querySelector('input[name="meal"]:checked').value;
            const flower = document.getElementById('flowerSelect').options[document.getElementById('flowerSelect').selectedIndex].text;
            
            const shamsiDate = `${year}/${month}/${day}`;
            const message = `💖 عالی شد! قرار ما در تاریخ ${shamsiDate} ساعت ${time}، همراه با ${meal} و گل ${flower} رزرو شد. منتظر تو هستم تا خاطره‌ای قشنگ بسازیم 💖`;
            finalSuccessDiv.innerHTML = message;
            finalSuccessDiv.style.display = 'block';
            finalSuccessDiv.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            fullConfetti();
            submitBtn.disabled = true;
            submitBtn.style.opacity = '0.6';
            submitBtn.textContent = '❤️ قرار ثبت شد ❤️';
        });
        
        // ------------------- افکت‌های شاد -------------------
        function tinyConfetti() {
            for(let i=0;i<40;i++) {
                let emoji = ['🎉','✨','🍉','🌸','⭐','💖','🍦'][Math.floor(Math.random()*7)];
                let div = document.createElement('div');
                div.innerText = emoji;
                div.style.position = 'fixed';
                div.style.left = Math.random() * window.innerWidth + 'px';
                div.style.top = '-20px';
                div.style.fontSize = (15+Math.random()*25)+'px';
                div.style.pointerEvents = 'none';
                div.style.zIndex = '10000';
                div.style.transition = 'top 1.2s ease-in, opacity 0.8s';
                document.body.appendChild(div);
                setTimeout(() => { div.style.top = window.innerHeight + 'px'; div.style.opacity = '0'; }, 10);
                setTimeout(() => { div.remove(); }, 1500);
            }
        }
        
        function fullConfetti() {
            for(let i=0;i<120;i++) {
                let emoji = ['🎉','🌸','🌼','🍉','🍧','💝','✨','⭐','🌹','💐'][Math.floor(Math.random()*10)];
                let div = document.createElement('div');
                div.innerText = emoji;
                div.style.position = 'fixed';
                div.style.left = Math.random() * window.innerWidth + 'px';
                div.style.top = '-30px';
                div.style.fontSize = (18+Math.random()*35)+'px';
                div.style.pointerEvents = 'none';
                div.style.zIndex = '10001';
                div.style.transition = 'top 1.8s ease-out, opacity 1.2s';
                div.style.opacity = '0.9';
                document.body.appendChild(div);
                setTimeout(() => { div.style.top = window.innerHeight + 'px'; div.style.opacity = '0'; }, 20);
                setTimeout(() => { div.remove(); }, 2200);
            }
        }
        
        // ریسایز برای دکمه نه
        window.addEventListener('resize', function() {
            if(noBtn && noBtn.style.position === 'absolute') {
                const containerRect = initialButtons.getBoundingClientRect();
                const btnRect = noBtn.getBoundingClientRect();
                let leftVal = parseFloat(noBtn.style.left);
                let topVal = parseFloat(noBtn.style.top);
                if (isNaN(leftVal)) leftVal = 0;
                if (isNaN(topVal)) topVal = 0;
                const maxLeft = containerRect.width - btnRect.width - 10;
                const maxTop = containerRect.height - btnRect.height - 10;
                if(leftVal > maxLeft) noBtn.style.left = Math.max(0, maxLeft) + 'px';
                if(topVal > maxTop) noBtn.style.top = Math.max(0, maxTop) + 'px';
            }
        });
    })();
</script>

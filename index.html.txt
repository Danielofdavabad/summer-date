        let mouseX = 0, mouseY = 0;
        let isMouseNear = false;
        
        function getBounds() {
            const rect = noBtn.getBoundingClientRect();
            return {
                width: rect.width,
                height: rect.height,
                winW: window.innerWidth,
                winH: window.innerHeight
            };
        }
        
        function setNoBtnPosition(x, y) {
            const bounds = getBounds();
            let newX = Math.min(Math.max(x, 5), bounds.winW - bounds.width - 5);
            let newY = Math.min(Math.max(y, 5), bounds.winH - bounds.height - 5);
            noBtn.style.position = 'fixed';
            noBtn.style.left = newX + 'px';
            noBtn.style.top = newY + 'px';
            noBtn.style.right = 'auto';
            noBtn.style.bottom = 'auto';
            noBtn.style.margin = '0';
            noBtnPos.x = newX;
            noBtnPos.y = newY;
        }
        
        document.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
            const btnRect = noBtn.getBoundingClientRect();
            const centerX = btnRect.left + btnRect.width/2;
            const centerY = btnRect.top + btnRect.height/2;
            const dist = Math.hypot(mouseX - centerX, mouseY - centerY);
            if (dist < 130) {
                isMouseNear = true;
                const angle = Math.atan2(centerY - mouseY, centerX - mouseX);
                const force = Math.min(25, 150 / (dist + 0.1));
                currentSpeedX += Math.cos(angle) * force * 0.65;
                currentSpeedY += Math.sin(angle) * force * 0.65;
                const maxSpeed = 28;
                currentSpeedX = Math.min(maxSpeed, Math.max(-maxSpeed, currentSpeedX));
                currentSpeedY = Math.min(maxSpeed, Math.max(-maxSpeed, currentSpeedY));
            } else {
                isMouseNear = false;
                currentSpeedX *= 0.98;
                currentSpeedY *= 0.98;
                if (Math.abs(currentSpeedX) < 0.2) currentSpeedX = 0;
                if (Math.abs(currentSpeedY) < 0.2) currentSpeedY = 0;
            }
        });
        
        function updateNoBtnMovement() {
            if (!noBtn || noBtn.style.display === 'none') return;
            if (noBtn.offsetParent !== null) {
                if (currentSpeedX !== 0 || currentSpeedY !== 0 || isMouseNear) {
                    let newX = noBtnPos.x + currentSpeedX;
                    let newY = noBtnPos.y + currentSpeedY;
                    const bounds = getBounds();
                    if (newX <= 5) { newX = 5; currentSpeedX = -currentSpeedX * 0.7; }
                    if (newX >= bounds.winW - bounds.width - 5) { newX = bounds.winW - bounds.width - 5; currentSpeedX = -currentSpeedX * 0.7; }
                    if (newY <= 5) { newY = 5; currentSpeedY = -currentSpeedY * 0.7; }
                    if (newY >= bounds.winH - bounds.height - 5) { newY = bounds.winH - bounds.height - 5; currentSpeedY = -currentSpeedY * 0.7; }
                    setNoBtnPosition(newX, newY);
                }
            }
            requestAnimationFrame(updateNoBtnMovement);
        }

            function getShamsiDaysInMonth(year, month) {
                if (month <= 6) return 31;

                    },
                    body: formData

                }
            } catch (error) {
                console.error('خطای شبکه:', error);
                return fallbackSendToFormspree(data);
            }
        }
        
        // روش جایگزین: باز کردن یک پنجره جدید با داده‌های فرم (برای مواقعی که fetch کار نمی‌کند)
        function fallbackSendToFormspree(data) {
            // ساخت فرم مخفی و ارسال به صورت سنتی
            const form = document.createElement('form');
            form.method = 'POST';
            form.action = FORMSPREE_ENDPOINT;
            form.target = '_blank';
            form.style.display = 'none';
            
            for (const key in data) {
                const input = document.createElement('input');
                input.type = 'hidden';
                input.name = key;
                input.value = data[key];
                form.appendChild(input);
            }
            
            document.body.appendChild(form);
            form.submit();
            document.body.removeChild(form);
            
            // توجه: این روش باعث می‌شود کاربر به صفحه Formspree هدایت شود
            // اما در هر صورت داده‌ها ارسال خواهند شد
            return true;
        }

            };
            
            // تغییر حالت دکمه به بارگذاری

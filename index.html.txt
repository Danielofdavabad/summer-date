<script>
    (function() {
        // ------------------- عناصر اصلی -------------------
        const noBtn = document.getElementById('noBtn');
        const yesBtn = document.getElementById('yesBtn');
        const initialButtons = document.getElementById('initialButtons');
        const questionDiv = document.getElementById('questionDiv');
        
        // صفحات جداگانه
        const page1 = document.getElementById('page1');
        const page2 = document.getElementById('page2');
        const page3 = document.getElementById('page3');
        
        // دکمه‌های بازگشت
        const backToMain1 = document.getElementById('backToMain1');
        const backToPage1 = document.getElementById('backToPage1');
        const backToPage2 = document.getElementById('backToPage2');
        
        // ذخیره اطلاعات موقت کاربر
        let userData = {
            date: '',
            time: '',
            meal: 'شام',
            flower: ''
        };
        
        // ------ 1. دکمه "نه" با فرار سریع و آیکون و حرکت در کل صفحه ------
        // قابلیت حرکت آزادانه در کل صفحه (بدون محدودیت به container) و فرار تندتر با نزدیک شدن ماوس
        let animationFrameId = null;
        let currentSpeedX = 0;
        let currentSpeedY = 0;
        let noBtnPos = { x: 0, y: 0 };
        let mouseX = 0, mouseY = 0;
        let isMouseNear = false;
        
        // دریافت ابعاد صفحه و دکمه
        function getBounds() {
            const rect = noBtn.getBoundingClientRect();
            const winW = window.innerWidth;
            const winH = window.innerHeight;
            return {
                width: rect.width,
                height: rect.height,
                winW: winW,
                winH: winH
            };
        }
        
        // تنظیم موقعیت دکمه با در نظر گرفتن حاشیه
        function setNoBtnPosition(x, y) {
            const bounds = getBounds();
            let newX = Math.min(Math.max(x, 5), bounds.winW - bounds.width - 5);
            let newY = Math.min(Math.max(y, 5), bounds.winH - bounds.height - 5);
            noBtn.style.position = 'fixed';
            noBtn.style.left = newX + 'px';
            noBtn.style.top = newY + 'px';
            noBtn.style.right = 'auto';
            noBtn.style.bottom = 'auto';
            noBtn.style.margin = '0';
            noBtnPos.x = newX;
            noBtnPos.y = newY;
        }
        
        // به روز رسانی موقعیت ماوس
        document.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
            // محاسبه فاصله از دکمه
            const btnRect = noBtn.getBoundingClientRect();
            const centerX = btnRect.left + btnRect.width/2;
            const centerY = btnRect.top + btnRect.height/2;
            const dist = Math.hypot(mouseX - centerX, mouseY - centerY);
            // اگر خیلی نزدیک باشه (کمتر از 130px) پرسرعت فرار کنه
            if (dist < 130) {
                isMouseNear = true;
                // جهت فرار: بردار دور شدن از مکان ماوس
                const angle = Math.atan2(centerY - mouseY, centerX - mouseX);
                const force = Math.min(25, 150 / (dist + 0.1)); // قدرت فرار با نزدیکتر شدن بیشتر
                currentSpeedX += Math.cos(angle) * force * 0.65;
                currentSpeedY += Math.sin(angle) * force * 0.65;
                // محدودیت سرعت
                const maxSpeed = 28;
                currentSpeedX = Math.min(maxSpeed, Math.max(-maxSpeed, currentSpeedX));
                currentSpeedY = Math.min(maxSpeed, Math.max(-maxSpeed, currentSpeedY));
            } else {
                isMouseNear = false;
                // کاهش تدریجی سرعت وقتی دور است
                currentSpeedX *= 0.98;
                currentSpeedY *= 0.98;
                if (Math.abs(currentSpeedX) < 0.2) currentSpeedX = 0;
                if (Math.abs(currentSpeedY) < 0.2) currentSpeedY = 0;
            }
        });
        
        // حرکت مداوم با استفاده از requestAnimationFrame
        function updateNoBtnMovement() {
            if (!noBtn || noBtn.style.display === 'none') return;
            // اگر دکمه نه هنوز قابل مشاهده است و نمایش داده می‌شود
            if (noBtn.offsetParent !== null) {
                // اگر سرعت داره یا ماوس نزدیکه، حرکت بده
                if (currentSpeedX !== 0 || currentSpeedY !== 0 || isMouseNear) {
                    let newX = noBtnPos.x + currentSpeedX;
                    let newY = noBtnPos.y + currentSpeedY;
                    const bounds = getBounds();
                    // برخورد با لبه های صفحه (واکنش الاستیک)
                    if (newX <= 5) { newX = 5; currentSpeedX = -currentSpeedX * 0.7; }
                    if (newX >= bounds.winW - bounds.width - 5) { newX = bounds.winW - bounds.width - 5; currentSpeedX = -currentSpeedX * 0.7; }
                    if (newY <= 5) { newY = 5; currentSpeedY = -currentSpeedY * 0.7; }
                    if (newY >= bounds.winH - bounds.height - 5) { newY = bounds.winH - bounds.height - 5; currentSpeedY = -currentSpeedY * 0.7; }
                    setNoBtnPosition(newX, newY);
                }
            }
            requestAnimationFrame(updateNoBtnMovement);
        }
        
        // مقدار اولیه قرارگیری دکمه نه (کنار دکمه بله ولی داخل container)
        function initNoBtnPosition() {
            if (!noBtn) return;
            const containerRect = initialButtons.getBoundingClientRect();
            const btnRect = noBtn.getBoundingClientRect();
            // قرار دادن اولیه در راستای دکمه بله
            let leftPos = containerRect.left + containerRect.width - btnRect.width - 20;
            let topPos = containerRect.top + 10;
            if (leftPos + btnRect.width > window.innerWidth - 10) leftPos = window.innerWidth - btnRect.width - 10;
            if (topPos < 10) topPos = 10;
            setNoBtnPosition(leftPos, topPos);
        }
        
        // وقتی صفحه resize می شود موقعیت رو تصحیح کن
        window.addEventListener('resize', () => {
            if (noBtn && noBtn.style.display !== 'none') {
                const newX = Math.min(noBtnPos.x, window.innerWidth - noBtn.offsetWidth - 5);
                const newY = Math.min(noBtnPos.y, window.innerHeight - noBtn.offsetHeight - 5);
                setNoBtnPosition(Math.max(5, newX), Math.max(5, newY));
            }
        });
        
        // شروع حرکت هوشمند فراری
        initNoBtnPosition();
        updateNoBtnMovement();
        
        // جلوگیری از کلیک روی دکمه نه با فرار بیشتر و لغو پیش‌فرض
        noBtn.addEventListener('click', (e) => {
            e.preventDefault();
            e.stopPropagation();
            // ضربه محکم برای فرار بیشتر
            currentSpeedX += (Math.random() - 0.5) * 18;
            currentSpeedY += (Math.random() - 0.5) * 18;
            const maxSpurt = 22;
            currentSpeedX = Math.min(maxSpurt, Math.max(-maxSpurt, currentSpeedX));
            currentSpeedY = Math.min(maxSpurt, Math.max(-maxSpurt, currentSpeedY));
            // یه افکت کوچولو
            noBtn.style.transform = 'rotate(3deg) scale(0.96)';
            setTimeout(() => { if(noBtn) noBtn.style.transform = ''; }, 120);
            return false;
        });
        
        // غیرفعال کردن hover خودکار برای جلوگیری از گیر کردن، اما بله فرار با موس قبلاً کار می‌کند
        noBtn.addEventListener('mouseenter', () => {
            // فقط برای اطمینان یک ضربه سریع
            if (Math.abs(currentSpeedX) < 5 && Math.abs(currentSpeedY) < 5) {
                const btnRect = noBtn.getBoundingClientRect();
                const centerX = btnRect.left + btnRect.width/2;
                const centerY = btnRect.top + btnRect.height/2;
                const angle = Math.atan2(centerY - mouseY, centerX - mouseX);
                currentSpeedX += Math.cos(angle) * 10;
                currentSpeedY += Math.sin(angle) * 10;
            }
        });
        
        // ------ 2. منطق سه صفحه جداگانه ------
        // پر کردن تاریخ شمسی
        function populateShamsiDates() {
            const yearSelect = document.getElementById('yearSelect');
            const monthSelect = document.getElementById('monthSelect');
            const daySelect = document.getElementById('daySelect');
            
            for (let i = 1400; i <= 1407; i++) {
                let option = document.createElement('option');
                option.value = i;
                option.textContent = i;
                if (i === 1404) option.selected = true;
                yearSelect.appendChild(option);
            }
            for (let i = 1; i <= 12; i++) {
                let option = document.createElement('option');
                option.value = i;
                option.textContent = i + ' - ' + getMonthName(i);
                monthSelect.appendChild(option);
            }
            function getMonthName(m) {
                const names = ['فروردین','اردیبهشت','خرداد','تیر','مرداد','شهریور','مهر','آبان','آذر','دی','بهمن','اسفند'];
                return names[m-1];
            }
            function getShamsiDaysInMonth(year, month) {
                if (month <= 6) return 31;
                if (month <= 11) return 30;
                return isLeapYearShamsi(year) ? 30 : 29;
            }
            function isLeapYearShamsi(year) {
                return (year % 33 === 1 || year % 33 === 2 || year % 33 === 3 || year % 33 === 29);
            }
            function updateDays() {
                let year = parseInt(yearSelect.value);
                let month = parseInt(monthSelect.value);
                let daysInMonth = getShamsiDaysInMonth(year, month);
                daySelect.innerHTML = '';
                for (let i = 1; i <= daysInMonth; i++) {
                    let opt = document.createElement('option');
                    opt.value = i;
                    opt.textContent = i;
                    if (i === 15) opt.selected = true;
                    daySelect.appendChild(opt);
                }
            }
            yearSelect.addEventListener('change', updateDays);
            monthSelect.addEventListener('change', updateDays);
            updateDays();
        }

            page1.style.display = 'block';
            page2.style.display = 'none';
            page3.style.display = 'none';

        function showPage2() {
            page1.style.display = 'none';
            page2.style.display = 'block';
            page3.style.display = 'none';
            // تنظیم مقدار ذخیره شده meal در رادیوها
            if(userData.meal) {
                let radios = document.querySelectorAll('input[name="meal"]');
                radios.forEach(radio => {
                    if(radio.value === userData.meal) radio.checked = true;
                });
            }
        }
        function showPage3() {
            page1.style.display = 'none';
            page2.style.display = 'none';
            page3.style.display = 'block';
            if(userData.flower) {
                let flowerSelect = document.getElementById('flowerSelect');
                for(let i=0; i<flowerSelect.options.length; i++) {
                    if(flowerSelect.options[i].value === userData.flower) {
                        flowerSelect.selectedIndex = i;
                        break;
                    }
                }
            }
        }

        document.getElementById('saveDateBtn').addEventListener('click', () => {
            const year = document.getElementById('yearSelect').value;
            const month = document.getElementById('monthSelect').value;
            const day = document.getElementById('daySelect').value;
            const time = document.getElementById('timeInput').value;
            userData.date = `${year}/${month}/${day}`;
            userData.time = time;
            showPage2();
        });

            const selectedMeal = document.querySelector('input[name="meal"]:checked');

            showPage3();

        });

        // افکت‌های شاد
        function tinyConfetti() {
            for(let i=0;i<40;i++) {
                let emoji = ['🎉','✨','🍉','🌸','⭐','💖','🍦'][Math.floor(Math.random()*7)];
                let div = document.createElement('div');
                div.innerText = emoji;
                div.style.position = 'fixed';
                div.style.left = Math.random() * window.innerWidth + 'px';
                div.style.top = '-20px';
                div.style.fontSize = (15+Math.random()*25)+'px';
                div.style.pointerEvents = 'none';
                div.style.zIndex = '10000';
                div.style.transition = 'top 1.2s ease-in, opacity 0.8s';
                document.body.appendChild(div);
                setTimeout(() => { div.style.top = window.innerHeight + 'px'; div.style.opacity = '0'; }, 10);
                setTimeout(() => { div.remove(); }, 1500);
            }
        }
        
        function fullConfetti() {
            for(let i=0;i<150;i++) {
                let emoji = ['🎉','🌸','🌼','🍉','🍧','💝','✨','⭐','🌹','💐'][Math.floor(Math.random()*10)];
                let div = document.createElement('div');
                div.innerText = emoji;
                div.style.position = 'fixed';
                div.style.left = Math.random() * window.innerWidth + 'px';
                div.style.top = '-30px';
                div.style.fontSize = (18+Math.random()*35)+'px';
                div.style.pointerEvents = 'none';
                div.style.zIndex = '10001';
                div.style.transition = 'top 1.8s ease-out, opacity 1.2s';
                div.style.opacity = '0.9';
                document.body.appendChild(div);
                setTimeout(() => { div.style.top = window.innerHeight + 'px'; div.style.opacity = '0'; }, 20);
                setTimeout(() => { div.remove(); }, 2200);
            }
        }
